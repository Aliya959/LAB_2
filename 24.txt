#define Etype int
using namespace std;
void SWAP(Etype  *a, Etype *b)
{
    Etype tmp = *a;
    *a = *b;
    *b = tmp;
}

int correct( int *arr, int size )
{
    while (--size > 0)
        if (arr[size - 1] > arr[size])
            return 0;
    return 1;
}
void shuffle( int *arr, int size )
{
    int i;
    for (i = 0; i < size; i++)
        SWAP(arr + i, arr + (rand() % size));
}
void bogoSort( int *arr, int size )
{
    while (!correct(arr, size))
        shuffle(arr, size);
}
